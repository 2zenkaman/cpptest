## Структура (struct):
```cpp
#include <iostream>
 
// Это - "шаблон" для создания новой переменной типа Point (точка с двумя координатами - x, y)
struct Point {
  int x, y;
};
 
int main() {
  Point p; // Создание нового объекта (новой точки), сейчас поля - рандомные значения, мусор
 
  // У меня это вывело "340341528 32765"
  std::cout << p.x << " " << p.y << std::endl;
 
  // Инициализация полей по отдельности, теперь вывод будет "10 20"
  p.x = 10;
  p.y = 20;
 
  // Инициализация списком - чуть короче и читабельнее, но надо помнить порядок полей
  Point center = {4, 7};
 
  // Тоже инициализация списком, другой синтаксис
  Point supremum{60, 47};
}
```
