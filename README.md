## Структура (`struct`):
```cpp
#include <iostream>
 
// Это - "шаблон" для создания новой переменной типа Point (точка с двумя координатами - x, y)
struct Point {
  int x, y;
};
 
int main() {
  Point p; // Создание нового объекта (новой точки), сейчас поля - рандомные значения, мусор
 
  // У меня это вывело "340341528 32765"
  std::cout << p.x << " " << p.y << std::endl;
 
  // Инициализация полей по отдельности, теперь вывод будет "10 20"
  p.x = 10;
  p.y = 20;
 
  // Инициализация списком - чуть короче и читабельнее, но надо помнить порядок полей
  Point center = {4, 7};
 
  // Тоже инициализация списком, другой синтаксис
  Point supremum{60, 47};
}
```

## Возврат результата из функции
```cpp
int square(int x); // возврат по значению - функция вернёт int
int a = square(9); // a = 81
 
// ---------------------------------------------------------------------------------------------------- //
 
int a, b;
scanf("%d %d", &a, &b); // scanf читает 2 числа из стандартного потока ввода (консоли)
// аргументы передаются через указатели, но логика та же, что и с ссылками

// ---------------------------------------------------------------------------------------------------- //
 
// возврат структуры Person
struct Person {
    int age;
    std::string name;
```

## Динамические массивы: операторы `new[]` и `delete[]`
```cpp
struct Point {
    int x, y;
};

int main() {
    // В C++ существует оператор new, который выделяет память в 
    // куче (где много свободной памяти) под новый объект данного типа

    // Мы дали оператору тип int, он выделил кусок памяти под int (4 байта) и вернул нам указатель на тот кусок
    int* count = new int;

    // Теперь с этой памятью можно работать
    *count = 10;

    // Оператор new[] выделяет блок памяти размером в (размер типа) * (количество элементов)
    
    // Оператор тоже вернул указатель, но теперь кусок памяти размером не на один int, а на 10 последовательных int
    // (4 байта) * (10 элементов) = 40 байт
    int* array = new int[10];

    // И с массивом тоже можно делать что хотим
    array[5] = 10;

    // Любая динамическая память, выделенная при помощи new, должна освобождаться при помощи оператора delete
    // Существует два варианта: один для массивов, другой — для единичных объектов

    delete count;   // Освобождение одной ячейки для a
    delete[] array; // Освобождение памяти массива

    // НО: сами указатели у нас до сих пор есть, их адреса мы нигде не меняли, а значит они до сих пор указывают
    // на ту самую память, которую мы взяли, а потом отдали

    // Так делать технически можно, но практически будет пизда, потому что память уже не наша
    *count = 10;

    // Можно выделять память под любые типы, даже пользовательские
    Point* center = new Point;
    Point* extremums = new Point[5];

    // Освобождение памяти выглядит аналогично
    delete center;
    delete[] extremums;
}
```
### Практика
Создать структуру `Point` с двуми полями `x`, `y`. Выделить память под массив точек размером 3 (вершины треугольника). Заполнить вручную или рандомно и посчитать периметр треугольника (сумму расстояний между всеми парами точек) и площадь треугольника. Вывести в терминал и освободить память.

## Перечисления `enum` и `enum class`
### `enum`
```cpp
// Enum - традиционный тип перечисления который пришел ещё из Си
// Он позволяет создать читаемые переменные, каждая из которых будет иметь особое целочисленное значение
enum HTTP_Status {
    OK = 200, // Мы можем задать значение произвольному элементу перечисления, по дефолту оно 0
    Created,  // А дальнейшие элементы автоматически имеют значение на 1 больше предыдущего
    Accepted, // Например это 202
    NonAuthoritativeInformation, // 203
    NoContent, // 204
    BadRequest = 400, // А здесь прыжок сразу на 400
    Unauthorized,     // И пошло: 401
    PaymentRequired   // 402
    // ............... //   // и так далее
};

int main() {
    // Проблема Enum в том, что все элементы перечислений глобальны и засоряют это поле видимости
    HTTP_Status s = OK; // Мы использовали StatusOK просто как глобальную переменную со значением 200

    // К тому же, могут случиться столкновения, когда какой-нибудь залупок определит переменную с таким же названием ещё раз
    // Особенно опасно это в простых перечислениях : enum Color {RED, GREEN, BLUE};

    // Также константы из перечислений Enum неявно конвертируются в целочисленный тип
    int s = OK;
}
```
### `enum class`
```cpp
// Поэтому в C++ ввели такую вещь как
enum class Day {
    Monday,   // Нумерация такая же - 0
    Tuesday, // 1
    Wednesday,  // 2
    Thursday, // и тд
    Friday,
    Saturday,
    Sunday
};

enum class Color {
    Red,
    Green,
    Blue
};

// Теперь Monday, Tuesday... не глобальные переменные, а поля класса Day

int main() {
    // Поэтому они не глобальные, а относятся к полю видимости класса Day:
    Day d = Day::Monday; // Похоже на std::cout, потому что cout - это объект в поле видимости пространства имён (namespace) std
    // И к ним нужно обращаться через ::

    // Это решило проблему засорения глобального пространства, теперь всё аккуратно лежит в "коробочке", к которой нужно явно обратиться

    // И теперь нельзя неявно привести его к целочисленным типам
    int day = Day::Tuesday; // ошибка

    // И это правильно: у нас не получится, например, сравнить цвет с днём недели
    if (Day::Tuesday == Color::Green) {}
    // Day::Tuesday это 1, и Color::Green это 1, но сравнивать их не имеет смысла. Такая вот защита от дурака
}
```
### Использование
```cpp
// Enum class удобно использовать в связке с switch, когда у нас есть ограниченный
// набор действий которые можно выполнять
enum class Operation {
    Add,
    Subtract,
    Multiply
};

// На основе переданного Operation функция выполняет одно действие
int calculate(int a, int b, Operation op) {
    switch (op)
    {
    case Operation::Add:
        return a + b;
        break;
    case Operation::Subtract:
        return a - b;
        break;
    case Operation::Multiply:
        return a * b;
        break;
    }
}

int main() {
    int result = calculate(2, 3, Operation::Add); // result = 5
}
```
## Побитовые операторы: `&`, `|`, `^`, `~`, `<<`, `>>`
Побитовые операторы работают с битовыми представлениям чисел

### `&` - битовое И
```cpp
// Битовое И работает с двумя числами (операндами)

int main() {
    // У нас есть два 8-битных числа (unsigned char)
    unsigned char a = 53, b = 176;

    // 53  -> 00110101
    // 176 -> 10110000
    // Мы смотрим на пары битов (по вертикали) и применяем к ним операцию
    // "И" возращает 1, если оба равны 1, 0 во всех других случаях

    // 53  -> 00110101
    // 176 -> 10110000
    //        ||||||||
    //        vvvvvvvv
    // 48  -> 00110000

    unsigned char result = a & b; // result = 48
}
```
